Code at a Glance

Less Code, More Intent

See how Nexa eliminates boilerplate. The same agent workflow that takes 60+ lines of Python takes just 15 lines of Nexa.

bugfixer.nx 
// Nexa — 15 lines, Harness built-in
agent BugFixer(issue: string) {
    system `Fix: #{issue}`

    context_policy {
        max_tokens: 100000,
        on_overflow: summarize
    }

    autoloop max_steps: 50 exit_when: "resolved" {
        try_agent {
            step();
        } catch_correction(e: ToolError) {
            reflect `Error: #{e}. Retry.`
        }
    }
}
